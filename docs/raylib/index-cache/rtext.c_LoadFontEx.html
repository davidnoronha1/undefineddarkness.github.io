Font LoadFontEx(const char *fileName, int fontSize, int *fontChars, int glyphCount){ Font font = { 0 }; // Loading file to memory unsigned int fileSize = 0; unsigned char *fileData = LoadFileData(fileName, &fileSize); if (fileData != NULL) { // Loading font from memory data font = LoadFontFromMemory(GetFileExtension(fileName), fileData, fileSize, fontSize, fontChars, glyphCount); UnloadFileData(fileData); } else font = GetFontDefault(); return font; }
