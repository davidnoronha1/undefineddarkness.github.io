void UntrackAudioBuffer(AudioBuffer *buffer){ ma_mutex_lock(&AUDIO.System.lock); { if (buffer->prev == NULL) AUDIO.Buffer.first = buffer->next; else buffer->prev->next = buffer->next; if (buffer->next == NULL) AUDIO.Buffer.last = buffer->prev; else buffer->next->prev = buffer->prev; buffer->prev = NULL; buffer->next = NULL; } ma_mutex_unlock(&AUDIO.System.lock); }
