void UnloadMesh(Mesh mesh){ // Unload rlgl mesh vboId data rlUnloadVertexArray(mesh.vaoId); if (mesh.vboId != NULL) for (int i = 0; i < MAX_MESH_VERTEX_BUFFERS; i++) rlUnloadVertexBuffer(mesh.vboId[i]); RL_FREE(mesh.vboId); RL_FREE(mesh.vertices); RL_FREE(mesh.texcoords); RL_FREE(mesh.normals); RL_FREE(mesh.colors); RL_FREE(mesh.tangents); RL_FREE(mesh.texcoords2); RL_FREE(mesh.indices); RL_FREE(mesh.animVertices); RL_FREE(mesh.animNormals); RL_FREE(mesh.boneWeights); RL_FREE(mesh.boneIds); }
