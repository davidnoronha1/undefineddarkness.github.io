int GetRandomValue(int min, int max){ if (min > max) { int tmp = max; max = min; min = tmp; } if ((unsigned int)(max - min) > (unsigned int)RAND_MAX) { TRACELOG(LOG_WARNING, "Invalid GetRandomValue() arguments, range should not be higher than %i", RAND_MAX); } return (rand()%(abs(max - min) + 1) + min); }
