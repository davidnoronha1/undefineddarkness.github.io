Image LoadImageRaw(const char *fileName, int width, int height, int format, int headerSize){ Image image = { 0 }; unsigned int dataSize = 0; unsigned char *fileData = LoadFileData(fileName, &dataSize); if (fileData != NULL) { unsigned char *dataPtr = fileData; unsigned int size = GetPixelDataSize(width, height, format); if (headerSize > 0) dataPtr += headerSize; image.data = RL_MALLOC(size); // Allocate required memory in bytes memcpy(image.data, dataPtr, size); // Copy required data to image image.width = width; image.height = height; image.mipmaps = 1; image.format = format; RL_FREE(fileData); } return image; }
