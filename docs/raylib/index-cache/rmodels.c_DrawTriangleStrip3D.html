void DrawTriangleStrip3D(Vector3 *points, int pointCount, Color color){ if (pointCount < 3) return; rlBegin(RL_TRIANGLES); rlColor4ub(color.r, color.g, color.b, color.a); for (int i = 2; i < pointCount; i++) { if ((i%2) == 0) { rlVertex3f(points[i].x, points[i].y, points[i].z); rlVertex3f(points[i - 2].x, points[i - 2].y, points[i - 2].z); rlVertex3f(points[i - 1].x, points[i - 1].y, points[i - 1].z); } else { rlVertex3f(points[i].x, points[i].y, points[i].z); rlVertex3f(points[i - 1].x, points[i - 1].y, points[i - 1].z); rlVertex3f(points[i - 2].x, points[i - 2].y, points[i - 2].z); } } rlEnd(); }
