void ImageDrawRectangleRec(Image *dst, Rectangle rec, Color color){ // Security check to avoid program crash if ((dst->data == NULL) || (dst->width == 0) || (dst->height == 0)) return; // Security check to avoid drawing out of bounds in case of bad user data if (rec.x < 0) { rec.width -= rec.x; rec.x = 0; } if (rec.y < 0) { rec.height -= rec.y; rec.y = 0; } if (rec.width < 0) rec.width = 0; if (rec.height < 0) rec.height = 0; int sy = (int)rec.y; int ey = sy + (int)rec.height; int sx = (int)rec.x; int bytesPerPixel = GetPixelDataSize(1, 1, dst->format); for (int y = sy; y < ey; y++) { // Fill in the first pixel of the row based on image format ImageDrawPixel(dst, sx, y, color); int bytesOffset = ((y*dst->width) + sx)*bytesPerPixel; unsigned char *pSrcPixel = (unsigned char *)dst->data + bytesOffset; // Repeat the first pixel data throughout the row for (int x = 1; x < (int)rec.width; x++) { memcpy(pSrcPixel + x*bytesPerPixel, pSrcPixel, bytesPerPixel); } } }
