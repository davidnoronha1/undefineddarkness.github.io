void ImageColorTint(Image *image, Color color){ // Security check to avoid program crash if ((image->data == NULL) || (image->width == 0) || (image->height == 0)) return; Color *pixels = LoadImageColors(*image); float cR = (float)color.r/255; float cG = (float)color.g/255; float cB = (float)color.b/255; float cA = (float)color.a/255; for (int y = 0; y < image->height; y++) { for (int x = 0; x < image->width; x++) { int index = y*image->width + x; unsigned char r = (unsigned char)(((float)pixels[index].r/255*cR)*255.0f); unsigned char g = (unsigned char)(((float)pixels[index].g/255*cG)*255.0f); unsigned char b = (unsigned char)(((float)pixels[index].b/255*cB)*255.0f); unsigned char a = (unsigned char)(((float)pixels[index].a/255*cA)*255.0f); pixels[index].r = r; pixels[index].g = g; pixels[index].b = b; pixels[index].a = a; } } int format = image->format; RL_FREE(image->data); image->data = pixels; image->format = PIXELFORMAT_UNCOMPRESSED_R8G8B8A8; ImageFormat(image, format); }
